#+begin_src sh
git clone --recurse-submodules https://github.com/andrewpeck/dotfiles ~/dotfiles

ln -s ~/dotfiles/home.nix ~/.config/nixpkgs/home.nix
#+end_src sh

#+begin_src sh
curl -L https://nixos.org/nix/install | sh

nix-channel --update
nix-shell '<home-manager>' -A install
home-manager switch
#+end_src sh
